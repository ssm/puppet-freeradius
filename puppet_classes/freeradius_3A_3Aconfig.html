class freeradius::config (
  Stdlib::Absolutepath $config_dir,
  String $config_owner,
  String $config_group,
  String $config_mode,
  Hash $users,
) {

  # Parameter
  $users_file = "${config_dir}/mods-config/files/authorize"

  # Manage users
  concat { $users_file:
    owner => $config_owner,
    group => $config_group,
    mode  => $config_mode,
  }
  $users.each |$user, $user_params| {
    $params = $user_params + { 'target' => $users_file }
    concat::fragment { $user:
      * => $params,
    }
  }
}
